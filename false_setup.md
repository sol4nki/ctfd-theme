run python inside your container terminal the command below will help you run bash inside your container

```docker
docker compose exec ctfd bash
```

once inside the container you will be able to run `python` by just typing python in the terminal


then just paste this inside python (make sure you ran `python` in the terminal)

```py
# copy paste this entire command
from CTFd import create_app
from CTFd.models import *

app = create_app()

exec("""
from random import choice, randint, sample
from CTFd.models import *

cats = ["WEB","PWN","CRYPTO","REV","FORENSICS","MISC","OSINT"]

with app.app_context():

    users = []
    for i in range(300):
        u = Users(
            name=f"user{i}",
            email=f"user{i}@test.local",
            password="Password123!"
        )
        db.session.add(u)
        users.append(u)

    db.session.commit()

    teams = []
    for i in range(200):
        t = Teams(name=f"team{i}")
        db.session.add(t)
        teams.append(t)

    db.session.commit()

    challenges = []

    for i in range(100):
        chal = Challenges(
            name=f"Challenge {i}",
            description=f"Autogenerated challenge {i}",
            category=choice(cats),
            value=randint(100,1000),
            state="visible",
            type="standard"
        )

        db.session.add(chal)
        db.session.flush()

        db.session.add(
            Flags(
                challenge_id=chal.id,
                type="static",
                content=f"flag{{challenge_{i}}}"
            )
        )

        challenges.append(chal)

    db.session.commit()

    users = Users.query.all()

    for user in users:
        solved = sample(challenges, randint(10,100))

        for chal in solved:
            db.session.add(
                Solves(
                    user_id=user.id,
                    challenge_id=chal.id,
                    team_id=user.team_id
                )
            )

    db.session.commit()

    print("DONE")
""")
```
